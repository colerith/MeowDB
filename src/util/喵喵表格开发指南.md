# 📋 MeowDB - 喵喵表格插件 需求文档（插件版）

## 一、项目定位

| 项目 | 说明 |
|------|------|
| **名称** | MeowDB / 喵喵表格 |
| **类型** | SillyTavern 插件（Extension） |
| **核心功能** | 基于特定模板（FM格式）的剧情数据管理与可视化 |
| **存储方式** | 聊天记录内嵌存储（`mes.meowDB` 或 `meowDB_data`），按聊天隔离 |
| **架构** | Vite + Vue3 + Pinia + TS，构建为单入口 `dist/index.js` + `dist/index.css` |
| **入口** | `src/index.ts`（插件启动与挂载） |
| **部署** | `manifest.json` 指向 `dist/index.js`/`dist/index.css` |

## 二、数据模型定义

### 2.1 核心数据结构

```typescript
interface MeowDBEntry {
  serial: string;        // "🐾喵喵摘要-001" 格式，强制001开始
  time: string;          // "yyyy-MM-dd 星期X | HH:mm 天气"
  nsfw: {
    current: number;     // 当前回合数
    max: number;         // 默认20
    details?: NSFWDetails; // 后续扩展
  };
  scene: {
    main: string;        // 主场景
    sub: string;         // 次级场景
    stayRounds: number;  // 停留回合
    topic: string;       // 本回合小剧场标题
  };
  plot: string;          // 150-300字，无换行，客观事实
  relations: CharacterRelation[]; // 在场角色卡片
  echoes: Echo[];        // 待回收承诺，上限10条
  archived: ArchivedNPC[]; // 离场NPC状态
  enigmas: Enigma[];     // 秘密情报
  seeds: Seed[];         // 伏笔，3-5条，花期机制
}

interface CharacterRelation {
  name: string;          // 角色名（必须包含Raven）
  gender: string;
  genitalStatus: string; // 性器官及状态
  identity: string;
  personality: string;
  sexExp: string;        // "对象(次数)"格式
  coordinate: string;    // 📍坐标
  clothing: string;      // 👗全套衣着（细化到头饰、内衣裤、鞋袜）
  action: string;        // ⚡实时动作
  bond: string;          // 与对方的当前羁绊
  favor: number;         // 好感值 ±0.0
  favorChange: string;   // 增幅原因简述
}

interface Echo {
  character: string;
  content: string;       // 待回收的关键承诺
}

interface ArchivedNPC {
  name: string;
  location: string;      // 当前去向/状态
  trigger: string;       // 唤醒契机
}

interface Enigma {
  content: string;       // 秘密内容
  progress: number;      // 揭露进度 N%
  related: string[];     // 关联角色
}

interface Seed {
  type: 'soul' | 'world'; // 灵魂(内在) / 世界(外部)
  name: string;
  bloom: number;         // 花期 N/5
  suspense: string;      // 悬念核
  driveLink: string;     // 驱动力链接
}
```

### 6.1 挂载位置代码（可视化面板）

```typescript
// 核心：找到 last_mes 并挂载在其下方
function mountPanel() {
  const lastMes = document.querySelector('.mes.last_mes');
  if (!lastMes) return;
  
  // 检查是否已挂载
  let panel = lastMes.querySelector('.meow-db-panel');
  if (panel) return panel;
  
  // 创建面板
  panel = document.createElement('div');
  panel.className = 'meow-db-panel';
  
  // 插入到消息内容区之后
  const mesContent = lastMes.querySelector('.mes_content');
  if (mesContent) {
    mesContent.after(panel);
  } else {
    lastMes.appendChild(panel);
  }
  
  return panel;
}
```

// 读取酒馆当前主题变量
function adaptToStTheme() {
  const root = document.documentElement;
  const computed = getComputedStyle(root);
  
  const themeVars = {
    accent: computed.getPropertyValue('--accent').trim(),
    bg: computed.getPropertyValue('--bg').trim(),
    text: computed.getPropertyValue('--text').trim(),
    border: computed.getPropertyValue('--border').trim(),
  };
  
  // 应用到meowDB面板
  const panel = document.querySelector('.meow-db-panel');
  if (panel) {
    Object.entries(themeVars).forEach(([key, val]) => {
      panel.style.setProperty(`--meow-${key}`, val);
    });
  }
}

### 6.3 数据存储实现

```typescript
// 写入当前Entry到最新消息
async function saveEntry(entry: MeowDBEntry) {
  const chat = window.SillyTavern.getContext().chat;
  const lastMsg = chat[chat.length - 1];
  
  if (!lastMsg.data) lastMsg.data = {};
  lastMsg.data.meowDB = entry;
  
  // 触发保存
  await window.SillyTavern.saveChat();
}

// 读取当前Entry
function loadEntry(): MeowDBEntry | null {
  const chat = window.SillyTavern.getContext().chat;
  
  // 从后往前找最新的Entry
  for (let i = chat.length - 1; i >= 0; i--) {
    const entry = chat[i].data?.meowDB;
    if (entry) return entry;
  }
  
  return null;
}
```
